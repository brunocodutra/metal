§ enumerate

template<typename start , typename size , typename stride = number<1>>
using metal::enumerate = typedef typename detail::_enumerate<start, size, stride>::type

#include <metal/number/enumerate.hpp>

Description

Generates a sequence of Numbers.

Usage

For any Numbers st, sz and sd

using result = metal::enumerate<st, sz, sd>;
Returns
: List
Semantics:
If sz is positive, then
using result = metal::numbers<
st{}, st{} + sd{}, ..., st{} + (sz{} - 1)*sd{}
>;
otherwise, if sz is negative, then
using result = metal::numbers<
st{}, st{} - sd{}, ..., st{} - (1 - sz{})*sd{}
>;
otherwise
using result = metal::numbers<>;

Example

IS_SAME(
metal::enumerate<metal::number<'a'>, metal::number<3>>,
metal::list<metal::number<'a'>, metal::number<'b'>, metal::number<'c'>>
);
IS_SAME(
metal::enumerate<metal::number<2>, metal::number<3>, metal::number<-5>>,
metal::list<metal::number<2>, metal::number<-3>, metal::number<-8>>
);
IS_SAME(
metal::enumerate<metal::number<2>, metal::number<-3>, metal::number<-5>>,
metal::list<metal::number<2>, metal::number<7>, metal::number<12>>
);

§ if_

template<typename cond , typename then_ , typename... else_>
using metal::if_ = typedef typename detail::_if_<cond, then_, else_...>::type

#include <metal/number/if.hpp>

Description

A multi-clause conditional expression.

Usage

For any Numbers num_1, ..., num_n and Values val_1, ..., val_n+1

using result = metal::if<num_1, val_1, ..., num_n, val_n, val_n+1>;
Returns
: Value
Semantics:
If num_i{} != false and num_j{} == false for all j < i, then
using result = val_i;
otherwise, if num_i{} == false for all i in [1, ..., n], then
using result = val_n+1;

Example

IS_SAME(metal::if_<metal::false_, int, metal::number<0>, int*, void>, void);
IS_SAME(metal::if_<metal::false_, int, metal::number<1>, int*, void>, int*);
IS_SAME(metal::if_<metal::true_, int, metal::number<1>, int*, void>, int);

§ pow

template<typename head , typename... tail>
using metal::pow = typedef typename detail::_pow<head, tail...>::type

#include <metal/number/pow.hpp>

Description

Computes the arithmetic exponentiation of Numbers.

Usage

For any Numbers num_1, ..., num_n

using result = metal::pow<num_1, ..., num_n>;
Precondition
: num_1{} ** ... ** num_m-1{} is nonzero for all negative num_m
Returns
: Number
Semantics:
using result = metal::number<num_1{} ** ... ** num_n{}>;
Note
Borrowing from Fortran, x ** y should be understood as x raised to the power of y.
Warning
x ** y is always null for y < 0 and |x| > 1 due to inherent limitations of integer arithmetic.

Example

using num = metal::pow<metal::number<-17>, metal::number<5>, metal::number<3>>;
IS_SAME(num, metal::number<-2862423051509815793>);
IS_SAME(
metal::is_invocable<metal::lambda<metal::pow>, metal::number<0>, num>,
metal::false_
);

See Also
